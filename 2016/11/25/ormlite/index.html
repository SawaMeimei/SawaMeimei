ormlite的使用

创建数据库以及表

新建一个类继承OrmLiteSqliteOpenHelper，实现里面的方法。


private static final String TABLE_NAME = "sqlite-test.db";

    private DatabaseOpenHelper(Context context) {
      //在构造里创建数据库
        super(context, TABLE_NAME, null, 2);
    }

    @Override
    public void onCreate(SQLiteDatabase database,
                         ConnectionSource connectionSource) {
        try {
          //建表语句，根据需要建表的Java类建立对应的表
            TableUtils.createTable(connectionSource, User.class);
            TableUtils.createTable(connectionSource, Article.class);
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }

    @Override
    public void onUpgrade(SQLiteDatabase database,
                          ConnectionSource connectionSource, int oldVersion, int newVersion) {
        try {
            TableUtils.dropTable(connectionSource, User.class, true);
            onCreate(database, connectionSource);
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }

对应的java对象

//使用DatabaseTable注解，tableName必须为tb_开头的tb_后为表名
@DatabaseTable(tableName = "tb_user")
public class User {
    //设置主键自增的id
    @DatabaseField(generatedId = true)
    private int id;

    //设置列名
    @DatabaseField(columnName = "name")
    private String name;

    @DatabaseField(columnName = "desc")
    private String desc;

    public User() {
    }

    public User(String name, String desc) {
        this.name = name;
        this.desc = desc;
    }

    public int getId() {
        return id;
    }

    public void setId(int id) {
        this.id = id;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getDesc() {
        return desc;
    }

    public void setDesc(String desc) {
        this.desc = desc;
    }

    @Override
    public String toString() {
        return "User{" +
                "id=" + id +
                ", name='" + name + '\'' +
                ", desc='" + desc + '\'' +
                '}';
    }
}

DAO的获取和CRUD操作

Dao dao = DatabaseOpenHelper.getHelper(context).getDao(clazz);

crud操作

public class DataDAO<T> {

    public Dao<T, Integer> dao;

    public DataDAO(Dao<T, Integer> dao) {
        this.dao = dao;
    }

    public boolean add(T t) {
        if (dao != null) {
            try {
                int i = dao.create(t);
                if (i == 1) {
                    return true;
                }
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }
        return false;
    }

    public boolean delete(T t) {
        if (dao != null) {
            try {
                int i = dao.delete(t);
                if (i == 1) {
                    return true;
                }
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }
        return false;
    }

    public boolean update(T t) {
        if (dao != null) {
            try {
                int i = dao.update(t);
                if (i == 1) {
                    return true;
                }
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }
        return false;
    }

    public List<T> queryAll() {
        if (dao != null) {
            try {
                return dao.queryForAll();
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }
        return new ArrayList<>();
    }
}

更多其他api还有queryById等，queryByRaw支持sql语句查询。
